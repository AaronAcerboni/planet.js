function startPoller(feedObj, name) {
  var poller = new Poller(feedObj, name);
  poller.init();
  all.push(poller);
}

function pollAll(){
  _.each(all, function(i) {
    i.pollNow();
  })
}

function get(id) {
  _.find(all, function(i) {
    return (i.id == id);
  })
}
