Activity initialization

For each resource found in aggregations.json create an associated activity

 
readConfig();

function readConfig() {
  fs.readFile("aggregations.json", function(err, data){
      var all = JSON.parse(data);

      for(i in all){
        var agg = all[i];
        for(j in agg.feeds){
          var feed = agg.feeds[j];
          switch(feed.collection.type){
            case "poll" :
              Activities.startPoller(feed, agg.name);
              break;
            case "pushed" :
              break;
          }
        }
      }
  });
}
